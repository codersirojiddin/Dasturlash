                    #include <iostream>
                    using namespace std;
                    class Calculator {
                        private:
                            double num1;
                            double num2;
                        
                        public:
                        //Konstruktor
                        Calculator(double a, double b){
                            num1 = a;
                            num2 = b;
                        }
                        
                        //Yig'indisi

                        double sum() {
                            return num1 + num2;
                        }

                        //ko'paytmasi
                        double mult() {
                            return num1 * num2;
                        }
                    };

                    int main() {
                        double a, b;
                        cout<<"1-son: ";
                        cin>>a;
                        cout<<"2-son: ";
                        cin>>b;

                        Calculator calc(a, b);
                        
                        cout<<"Yig`indi: "<<calc.sum()<<endl;
                        cout<<"Ko`paytma: "<<calc.mult()<<endl;

                        return 0;
                    }
